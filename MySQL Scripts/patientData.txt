use produhdb;
CREATE TABLE patientData (
  patientId INT PRIMARY KEY,
  firstName VARCHAR(50),
  middleName VARCHAR(50),
  lastName VARCHAR(50),
  address VARCHAR(100),
  city VARCHAR(50),
  state VARCHAR(50),
  telephone VARCHAR(20),
  email VARCHAR(100),
  dateOfBirth DATE,
  healthInsuranceNumber VARCHAR(50),
  emergencyContactNumber VARCHAR(20)
    ) default charset = latin1;
